# 即梦图片生成 MCP 服务器环境配置示例
# 复制此文件为 .env 并填入实际值

# 即梦API服务器地址 (默认为本地地址)
JIMENG_API_BASE=http://localhost:8001

# 即梦API会话ID (必须设置)
# 从即梦API获取的认证session_id
JIMENG_SESSION_ID=your_session_id_here

# 可选配置
# 默认模型
DEFAULT_MODEL=jimeng-3.0

# 默认图片尺寸
DEFAULT_WIDTH=1024
DEFAULT_HEIGHT=1024

# 默认采样强度
DEFAULT_SAMPLE_STRENGTH=0.5

# API请求超时时间 (秒)
REQUEST_TIMEOUT=120

# 腾讯云对象存储配置 (可选)
# 如需自动上传图片到腾讯云COS，请配置以下参数

# 腾讯云API密钥ID
TENCENT_CLOUD_SECRET_ID=your_tencent_cloud_secret_id

# 腾讯云API密钥Key
TENCENT_CLOUD_SECRET_KEY=your_tencent_cloud_secret_key

# COS存储桶地域 (默认: ap-guangzhou)
# 常用地域: ap-beijing(北京), ap-shanghai(上海), ap-guangzhou(广州), ap-singapore(新加坡)
TENCENT_COS_REGION=ap-guangzhou

# COS存储桶名称 (默认: jimeng-images)
TENCENT_COS_BUCKET=jimeng-images

# 自定义域名 (可选，如不设置则使用默认COS域名)
TENCENT_COS_DOMAIN=your-custom-domain.com
